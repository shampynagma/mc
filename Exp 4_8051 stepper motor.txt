Stepper motor 8051

#include<reg51.h>

void delay() {
    TMOD=0x10;
    TH1=0x00;
    TL1=0x00;
    TR1=1;
    while(TF1==0);
    TR1=0;
    TF1=0;
}

void main() {
    unsigned int i;
    unsigned char arr[4]={0x01, 0x02, 0x04, 0x08};
    while(1) {
        for(i=0; i<4; i++) {
            P2=arr[i];
            delay();
        }
    }
}
